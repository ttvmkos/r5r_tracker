    "loader.cpp"
    "loader.h"
)

    "advapi32.lib"
    "bcrypt.lib"

    "wldap32.lib"
    "ws2_32.lib"
    "Rpcrt4.lib"

    "vpc"
    "tier0"

    "libdetours"
)

    "/STACK:8000000" # Match game executable stack reserve size
)
